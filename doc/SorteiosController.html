def create

  end
end

def update
  @sorteio = Sorteio.find(params[:id])
  if @sorteio.update_attributes(params[:sorteio])
    redirect_to @sorteio, notice: 'Sorteio atualizado.'
  else
    render action: "edit"
  end
end
